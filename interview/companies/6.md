### Что выведется

```js
'use strict'
var obj = {
    i: 10,
    b: () => console.log(this.i, this),
    c: function() {
        console.log(this.i, this)
    }
}

obj.b()
obj.c()
```

### Что выведется

```js
var test = (function() {
    var value = 1;
    function setValue(newValue) {
        value = newValue
    }
    return {
        value: value,
        setValue: setValue
    }
})();
test.setValue(2);
console.log(test.value)
```
