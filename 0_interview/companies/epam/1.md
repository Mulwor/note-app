```js
import React, { useState, useEffect } from "react";
const API_URL = "https://test.com";

export function SearchResult() {
  const [searchParam, setSearchParam] = useState(null);
  const [dropdownArr, setDropdownArr] = useState(null);
  const inputRef = useRef(null);

  const fetchData = (searchParam) =>
    fetch(`${API_URL}?s=${searchParams}`)
      .then((response) => response.json())
      .then((data) => setDropdownArr(data))
      .catch((error) => console.log(error));

  const onChange = (event) => {
    setSearchQuery(event.target.value);
  };

  useEffect(() => {
    inputRef.current.focus();
  }, []);

  return (
    <div>
      <input onChange={onChange} ref={inputRef} />
      <select>
        {dropdownArr.map((item, index) => {
          <option key={`${index}_result_data`}>{item}</option>;
        })}
      </select>
    </div>
  );
}
```

Task #2

- Remove all duplicate values form an array.
- Sort the array value in ascending order
- Find the sum of elements that are located at even indices in an array

```js
const initialArray = [10, 30, 20, 10, 40, 50, 20, 60, 40];

const getUniqArray = (arr) => {};
const getSortedArray = (arr) => {};
const getSum = (arr) => {};

const uniqArray = getUniqArray(initialArray);
const sortedArray = getSortedArray(initialArray);
const sum = getSum(initialArray);

console.log("#1 ", uniqArray);
console.log("#2 ", sortedArray);
console.log("#3 ", sum);
```

Answer

```js
const getUniqArray = (arr) => {
  return [...new Set(array)];
};

const getSortedArray = (arr) => {
  return arr.sort((a, b) => a - b);
  // toSorted - if we need avoid mutation
};

const getSum = (arr) => {
  return arr.reduce((accum, value) => accum + value, 0);
};
```
