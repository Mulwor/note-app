## Code Review

```js
import React, { useState, useEffect } from 'react';
import loadLocation from 'tools/loadLocation';

export default BookingMap = (props: any) => {
  const [spot, updateSpot] = useState();
  const [location, updateLocation] = useState({});

  useEffect(() => {
    loadLocation(props.locationId).then(location => updateLocation(location))
  });

  const availableSpotTime = props.spots.map(s => {
    const availableTimeSlot = s.timeslots.find(t => t.isAvailable);

    return {
      [s.id]: availableTimeSlot
    }
  });

  const selectSpot = id => updateSpot(id);

  return (
    <>
      <header>Spots for {location.name}</header>
      {
        props.spots.map(s => {
          return <div onClick={() => selectSpot(s.id)} className={`spot ${spot == s.id ? 'active' : ''}`}>{s.id} at {availableSpotTime[s.id]}</div>
        });
      }
    </>
  );
}
```

Задача на знание базовых вещей в React и способность их объяснить внятным языком. Также обзор культуры написания кода (конвенции и соглашения, читабельность). Способность донести обратную связь по увиденному коду и способность обосновать, что решение плохое.
